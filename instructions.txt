# Terminal 1 — Backend
cd backend

uvicorn app.main:app --reload --port 8000

cd frontend

# Stop containers first
docker-compose down

# Rebuild with the fix
docker-compose up --build


# Only use --build when:

docker-compose up --build   # 1. First time ever
docker-compose up --build   # 2. Changed requirements.txt
docker-compose up --build   # 3. Changed package.json
docker-compose up --build   # 4. Changed Dockerfile
docker-compose up --build   # 5. Something is broken and
                            #    you want a clean rebuild

# For everything else (daily code changes):
docker-compose up           # ← no --build needed


# If something is really broken and you want 100% fresh:

# Stop and remove everything
docker-compose down --volumes --rmi all

# Then rebuild from scratch
docker-compose up --build

# ⚠️ This deletes cached layers, will be slow again
